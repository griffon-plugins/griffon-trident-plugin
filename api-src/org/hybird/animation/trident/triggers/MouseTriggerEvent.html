16 package org.hybird.animation.trident.triggers;
17 
18 /**
19  * Mouse Enter/Exit/Press/Release/Click events
20  *
21  * @author Chet
22  */
23 public class MouseTriggerEvent extends TriggerEvent {
24     /**
25      * Event fired when mouse enters
26      */
27     public static final MouseTriggerEvent ENTER = new MouseTriggerEvent("Entered");
28     /**
29      * Event fired when mouse exits
30      */
31     public static final MouseTriggerEvent EXIT = new MouseTriggerEvent("Exit");
32     /**
33      * Event fired when mouse button is pressed
34      */
35     public static final MouseTriggerEvent PRESS = new MouseTriggerEvent("Press");
36     /**
37      * Event fired when mouse button is released
38      */
39     public static final MouseTriggerEvent RELEASE = new MouseTriggerEvent("Release");
40     /**
41      * Event fired when mouse is clicked
42      */
43     public static final MouseTriggerEvent CLICK = new MouseTriggerEvent("Click");
44 
45     /**
46      * Protected constructor; this helps ensure type-safe use of pre-defined
47      * TriggerEvent objects.
48      */
49     private MouseTriggerEvent(String name) {
50         super(name);
51     }
52 
53     /**
54      * This method finds the opposite of the current event.: <BR/>
55      * ENTER -> EXIT <BR/>
56      * EXIT -> ENTER <BR/>
57      * PRESS -> RELEASE <BR/>
58      * RELEASE -> PRESS <BR/>
59      * Note that CLICK has no obvious opposite so it simply returns CLICK (this
60      * method should probably not be called for that case).
61      */
62     public TriggerEvent getOppositeEvent() {
63         if (this == MouseTriggerEvent.ENTER) {
64             return MouseTriggerEvent.EXIT;
65         } else if (this == MouseTriggerEvent.EXIT) {
66             return MouseTriggerEvent.ENTER;
67         } else if (this == MouseTriggerEvent.PRESS) {
68             return MouseTriggerEvent.RELEASE;
69         } else if (this == MouseTriggerEvent.RELEASE) {
70             return MouseTriggerEvent.PRESS;
71         }
72         // Possible to reach here for REPEAT action (but probably should not
73         // have been called with this event)
74         return this;
75     }
76 }
