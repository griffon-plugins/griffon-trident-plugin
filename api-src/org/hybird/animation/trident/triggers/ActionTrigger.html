16 package org.hybird.animation.trident.triggers;
17 
18 import org.pushingpixels.trident.Timeline;
19 
20 import java.awt.event.ActionEvent;
21 import java.awt.event.ActionListener;
22 import java.lang.reflect.Method;
23 
24 /**
25  * ActionTrigger handles action events and starts the timeline when actions
26  * occur. For example, to have anim start when a button is clicked, one might
27  * write the following:
28  * <p>
29  * <pre>
30  * ActionTrigger trigger = ActionTrigger.addTrigger (button, anim);
31  * </pre>
32  *
33  * @author Chet
34  * @author Rï¿½my Rakic - Modifications for Trident and javadocs
35  */
36 public class ActionTrigger extends Trigger implements ActionListener {
37 
38     /**
39      * Creates an ActionTrigger and adds it as a listener to object.
40      *
41      * @param object   an object that will be used as an event source for this
42      *                 trigger. This object must have the method addActionListener.
43      * @param timeline the Timeline that will start when the event occurs
44      *
45      * @return ActionTrigger the resulting trigger
46      *
47      * @throws IllegalArgumentException if object has no <code>addActionListener()</code>
48      */
49     public static ActionTrigger addTrigger(Object object, Timeline timeline) {
50         ActionTrigger trigger = new ActionTrigger(timeline);
51         try {
52             Method addListenerMethod = object.getClass().getMethod("addActionListener", ActionListener.class);
53             addListenerMethod.invoke(object, trigger);
54         } catch (Exception e) {
55             throw new IllegalArgumentException("Problem adding listener to object: " + e);
56         }
57         return trigger;
58     }
59 
60     /**
61      * Creates an ActionTrigger that will start the timeline upon receiving any
62      * ActionEvents. It should be added to any suitable object with an
63      * addActionListener method.
64      *
65      * @param timeline the Timeline that will start when the event occurs
66      */
67     public ActionTrigger(Timeline timeline) {
68         super(timeline);
69     }
70 
71     /**
72      * Called by an object generating ActionEvents to which this trigger was
73      * added as an ActionListener. This starts the Timeline.
74      */
75     public void actionPerformed(ActionEvent ae) {
76         fire();
77     }
78 }
